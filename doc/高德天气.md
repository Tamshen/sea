# 天气查询

## OpenAPI Specification

```yaml
openapi: 3.0.1
info:
  title: ''
  description: ''
  version: 1.0.0
paths:
  /v3/weather/weatherInfo:
    get:
      summary: 天气查询
      deprecated: false
      description: ''
      tags:
        - 开发指南/高级 API 文档/天气查询
      parameters:
        - name: city
          in: query
          description: >-
            城市编码，输入城市的adcode，adcode信息可参考[城市编码表](https://lbs.amap.com/api/webservice/download)
          required: false
          example: '110101'
          schema:
            type: string
        - name: extensions
          in: query
          description: 气象类型  可选值：base/all  base:返回实况天气  all:返回预报天气
          required: false
          schema:
            type: string
        - name: output
          in: query
          description: 返回格式  可选值：JSON,XML
          required: false
          schema:
            type: string
      responses:
        '200':
          description: ''
          content:
            application/json:
              schema:
                type: object
                properties:
                  status:
                    type: string
                    description: 返回状态  值为0或1  1：成功；0：失败
                  count:
                    type: number
                    description: 返回结果总数目
                  info:
                    type: string
                    description: 返回的状态信息
                  infocode:
                    type: string
                    description: 返回状态说明,10000代表正确
                  lives:
                    type: array
                    items:
                      type: object
                      properties:
                        province:
                          type: string
                          description: 省份名
                        city:
                          type: string
                          description: 城市名
                        adcode:
                          type: string
                          description: 区域编码
                        weather:
                          type: string
                          description: 天气现象（汉字描述）
                        temperature:
                          type: number
                          description: 实时气温，单位：摄氏度
                        winddirection:
                          type: string
                          description: 风向描述
                        windpower:
                          type: string
                          description: 风力级别，单位：级
                        humidity:
                          type: number
                          description: 空气湿度
                        reporttime:
                          type: string
                          description: 数据发布的时间
                        forecast:
                          type: object
                          properties:
                            city:
                              type: string
                              description: 城市名称
                            adcode:
                              type: string
                              description: 城市编码
                            province:
                              type: string
                              description: 省份名称
                            reporttime:
                              type: string
                              description: 预报发布时间
                            casts:
                              type: array
                              items:
                                type: object
                                properties:
                                  date:
                                    type: string
                                    description: 日期
                                  week:
                                    type: string
                                    description: 星期几
                                  dayweather:
                                    type: string
                                    description: 白天天气现象
                                  nightweather:
                                    type: string
                                    description: 晚上天气现象
                                  daytemp:
                                    type: number
                                    description: 白天温度
                                  nighttemp:
                                    type: number
                                    description: 晚上温度
                                  daywind:
                                    type: string
                                    description: 白天风向
                                  nightwind:
                                    type: string
                                    description: 晚上风向
                                  daypower:
                                    type: string
                                    description: 白天风力
                                  nightpower:
                                    type: string
                                    description: 晚上风力
                              description: 预报数据list结构，元素cast,按顺序为当天、第二天、第三天的预报数据
                          description: 预报天气信息数据
                      description: 实况天气数据信息
              example:
                status: '1'
                count: '1'
                info: OK
                infocode: '10000'
                lives:
                  - province: 北京
                    city: 东城区
                    adcode: '110101'
                    weather: 阴
                    temperature: '8'
                    winddirection: 南
                    windpower: ≤3
                    humidity: '20'
                    reporttime: '2022-03-22 16:10:59'
          headers: {}
          x-apifox-name: 成功
      security: []
      x-apifox-folder: 开发指南/高级 API 文档/天气查询
      x-apifox-status: released
      x-run-in-apifox: https://app.apifox.com/web/project/759159/apis/api-14675765-run
components:
  schemas: {}
  securitySchemes:
    key:
      type: apikey
      in: query
      name: key
servers:
  - url: https://restapi.amap.com
    description: 正式环境
security: []

```
